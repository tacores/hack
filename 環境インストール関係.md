```sh
# クリップボードが機能しない場合に実行
pkill -f 'VBoxClient --clipboard'
/usr/bin/VBoxClient --clipboard

# 永続化エイリアス
echo "alias resetclip=\"pkill -f 'VBoxClient --clipboard'; /usr/bin/VBoxClient --clipboard\"" >> ~/.zshrc
source ~/.zshrc
```
